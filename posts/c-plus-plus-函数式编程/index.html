#include <iostream>
#include <vector>

double average_score(const std::vector<int>& scores) {
    int sum = 0;
    for (int score : scores) {
        sum += score;
    }
    return sum / (double)scores.size();
}

int main() {
    std::vector<int> iv{1, 2, 3, 4, 5};
    std::cout << average_score(iv) << std::endl;
    return 0;
}

#include <iostream>
#include <vector>
#include <numeric>

double average_score(const std::vector<int>& scores) {
    return std::accumulate(
        scores.cbegin(), scores.cend(),
        0
    ) / (double)scores.size();
}

int main() {
    std::vector<int> iv{1, 2, 3, 4, 5};
    std::cout << average_score(iv) << std::endl;
    return 0;
}

#include <iostream>

double scores_product(const std::vector<int>& scores) {
    return std::accumulate(
        scores.cbegin(), scores.cend(),
        1,
        std::multiplies<int>()
    );
}

折叠

为了计算某些东西，通常需要一次处理集合中的一个条目。结果可能是一个简单的和或积（就像前面），也可能是包含特定值或符合预定谓词的条目的数目。结果可能更复杂，比如，包含原集合中部分元素的新集合（过滤，filter），或元素重新排序的新集合（用于排序或分区）。

std::accumulate 算法是折叠（folding）（或精减（reduction））的一种实现。这是一个高阶函数，它提供了对递归结构，如向量、列表和树等的遍历处理，并允许逐步构建自己需要的结果。在前一个例子中，使用折叠计算分数集合的累加和。 std::accumulate 算法首先计算传递给它的初始值和集合中第一个元素的和，结果再与集合中下一个元素相加，以此类推，一直重复到集合结束。

通常折叠不要求实际参数与传递给它的二进制函数有相同的类型。折叠接受一个集合，其中包含 T 类型的条目、R 类型的初始值（没必要与 T 是相同的类型）和一个函数 f:(R, T)->R 。对初始值和集合的第一个元素调用给定的函数。结果与集合中的第二个元素在传递给函数 f。一直重复到集合中所有元素处理完毕。算法返回一个 R 类型的值，这个值也是函数 f 最后一次调用的返回值。

#include <iostream>
#include <string>
#include <numeric>

int f(int previous_count, char c) {
    return (c != '\n') ? previous_count
        : previous_count + 1;
}

int count_lines(const std::string& s) {
    return std::accumulate(
        s.cbegin(), s.cend(),
        0,
        f
    );
}

int main() {
    std::string s{"aaa\nbbb\nccc\n"};
    std::cout << count_lines(s) << std::endl;
    std::string raw_str = R"(aaa
        bbb
        ccc)";
    std::cout << count_lines(raw_str) << " new line in raw_str" << std::endl;
    return 0;
}

删除字符串空白符

STL 没有去除字符串首尾空白的函数。但它提供了一些算法可以实现这样的功能。

#include <iostream>
#include <string>
#include <algorithm>

bool is_not_space(char ch) {
    return ch != ' ';
}

std::string trim_left(std::string s) {
    s.erase(s.begin(),
            std::find_if(s.begin(), s.end(), is_not_space));
    return s;
}

std::string trim_right(std::string s) {
    s.erase(std::find_if(s.rbegin(), s.rend(), is_not_space).base(),
            s.end());
    return s;
}

std::string trim(std::string s) {
    return trim_left(trim_right(std::move(s)));
}

int main() {
    std::string s{"   aaa   "};
    std::cout << s << "bbb" << std::endl;
    std::cout << trim(s) << "bbb" << std::endl;
    return 0;
}

基于谓词分割集合

std::partition 和它的变体 std::stable_partition , 两个算法接受一个集合和一个谓词。它们对原集合中的元素进行重排，把符合条件的与不符合条件的分开。符合谓词条件的元素移动到集合的前面，不符合条件的元素移动到集合的后面。算法返回一个迭代器，指向第二部分的第一个元素（不符合谓词条件的第一个元素）。

std::stable_partition 可以保持集合中原来的顺序。

#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>

std::vector<int>& insert(std::vector<int>& res, int cur) {
    auto it = std::find_if(res.cbegin(), res.cend(), [cur](int val) {
        return val > cur;
        })  ;
    res.insert(it, cur);
    return res;
}

int main() {
    std::vector<int> iv{6, 5, 4, 3, 2, 1};
    std::vector<int> sorted_v;
    sorted_v = std::accumulate(iv.cbegin(), iv.cend(), sorted_v, insert);
    for (auto i : sorted_v) {
        std::cout << i << " ";
    }
    std::cout << std::endl;
    return 0;
}
